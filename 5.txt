#!"C:\xampp\perl\bin\perl.exe"

use CGI':standard';
use DBI;

print "Content-type: text/html; \n\n";

$db = DBI->connect("DBI:mysql:test;", 'root');

$q = $db->prepare("select * from info");
$q->execute();

print "<table border=1px>";

while (($n, $a) = $q->fetchrow()) {
    print "<tr> <td>$n</td> <td>$a</td> </tr>";
}

print "</table>";

exit(0);
